    **_run_coverage(project_root: Path) -> dict:**
    - Call subprocess.run(["python3", "-m", "coverage", "run", "-m", "pytest", "--tb=no", "-q"], capture_output=True, text=True, cwd=project_root)
    - Check if first call's stderr contains "No module named" — if so, return {}
    - Call subprocess.run(["python3", "-m", "coverage", "json", "-o", str(project_root / "coverage.json")], capture_output=True, text=True, cwd=project_root)
    - Read project_root / "coverage.json", parse with json.loads
    - Build dict: {path: info["summary"]["percent_covered"] for path, info in data.get("files", {}).items()}
    - Cleanup: (project_root / "coverage.json").unlink(missing_ok=True)
    - Return the dict. Wrap entire function in try/except Exception returning {} for any unexpected failure.

    **_check_docstrings(py_file: Path) -> list[str]:**
    - try: tree = ast.parse(py_file.read_text(encoding="utf-8"))
    - except SyntaxError: return []
    - missing = []
    - for node in ast.walk(tree): if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)): if ast.get_docstring(node) is None: missing.append(f"{py_file.name}::{node.name} (line {node.lineno})")
    - return missing

    **_check_oss_files(project_root: Path) -> dict[str, bool]:**
    - return {name: (project_root / name).exists() for name in _OSS_FILES}

    **_run_flake8(scripts_dir: Path) -> list[dict]:**
    - result = subprocess.run(["python3", "-m", "flake8", "--select=F", str(scripts_dir)], capture_output=True, text=True)
    - if result.returncode not in (0, 1): return []
    - Parse each line of result.stdout: split(":", 3), extract file (Path(parts[0]).name), line, col, code_msg. code = code_msg.strip().split()[0] if code_msg.strip() else ""
    - Return list of {"file": ..., "line": ..., "col": ..., "code": ..., "message": code_msg.strip()}

    **_branch_complexity(py_file: Path) -> list[dict]:**
    - try: tree = ast.parse(py_file.read_text(encoding="utf-8"))
    - except SyntaxError: return []
    - hotspots = []
    - for node in ast.walk(tree): if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
      - count = sum(1 for n in ast.walk(node) if isinstance(n, _BRANCH_TYPES))
      - if count >= COMPLEXITY_THRESHOLD: hotspots.append({"file": py_file.name, "name": node.name, "line": node.lineno, "branches": count})
    - return hotspots
