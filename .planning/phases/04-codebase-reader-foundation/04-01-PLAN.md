        ".pytest_cache", "htmlcov", ".coverage", ".ruff_cache",

        ".pyc", ".pyo", ".so", ".dylib", ".dll", ".exe", ".bin",

    })
    DEFAULT_CREDENTIAL_FILES: frozenset = frozenset({
        ".env", ".env.local", ".env.production", ".env.staging",
        "credentials.json", "service_account.json",
    })
    CREDENTIAL_EXTENSIONS: frozenset = frozenset({
        ".pem", ".key", ".p12", ".pfx", ".jks",

    TOKEN_BUDGET_DEFAULT: int = 40_000
    MAX_FILE_BYTES: int = 100_000

    **Secret patterns (copy from research, do NOT import from generate_pretty.py):**
    ```python
    _SECRET_PATTERNS = [
        re.compile(r"AIza[a-zA-Z0-9_-]{30,}"),
        re.compile(r"sk-or-v1-[a-zA-Z0-9_-]{30,}"),
        re.compile(r"sk-[a-zA-Z0-9]{20,}"),
        re.compile(r"ghp_[a-zA-Z0-9]{36,}"),
        re.compile(r"(?i)(?:api[_-]?key|token|secret|password)\s*[:=]\s*['\"]?([A-Za-z0-9_/+=]{16,})['\"]?"),
    ]
    ```

    **Helper functions:**
    - `_is_binary(path: Path) -> bool` — read first 1024 bytes, return True if b"\x00" found or exception
    - `_is_credential_file(path: Path) -> bool` — check name in DEFAULT_CREDENTIAL_FILES, suffix in CREDENTIAL_EXTENSIONS, name starts with "service_account" and suffix == ".json"
    - `_redact_content(text: str) -> str` — apply all _SECRET_PATTERNS via pat.sub("[REDACTED]", text)
    - `_build_noise_filter(root: Path) -> pathspec.PathSpec | None` — load .gitignore if _PATHSPEC_OK and file exists
    - `_read_file_safe(path: Path, max_bytes: int = MAX_FILE_BYTES) -> str` — skip binary, read utf-8 errors="ignore"
    - `_should_skip_dir(dirname: str) -> bool` — return dirname in DEFAULT_SKIP_DIRS
    - `_should_skip_file(path: Path, gitignore_spec) -> bool` — check extension in DEFAULT_SKIP_EXTENSIONS, credential check, gitignore match

    **Public API `read_codebase(root: Path, token_budget: int = TOKEN_BUDGET_DEFAULT) -> dict`:**
    - Walk directory with os.walk, prune dirs matching DEFAULT_SKIP_DIRS in-place
    - For each file: skip if _should_skip_file, skip if _is_binary
    - Read content via _read_file_safe, apply _redact_content
    - Accumulate into files list with relative paths
    - Build summary_text by concatenating file contents wrapped in `<file path="rel_path">content</file>` delimiters
    - Return CodebaseReport dict (stub layers/connections as empty lists for now — Plan 02 fills these)
    - Token budget logic is a placeholder pass-through for now (Plan 02 implements it fully)
